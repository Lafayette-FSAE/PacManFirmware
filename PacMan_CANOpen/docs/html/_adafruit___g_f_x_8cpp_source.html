128 void Adafruit_GFX::writeLine(int16_t x0, int16_t y0, int16_t x1, int16_t y1,
129  uint16_t color) {
130 #if defined(ESP8266)
131  yield();
132 #endif
133  int16_t steep = abs(y1 - y0) > abs(x1 - x0);
134  if (steep) {
135  _swap_int16_t(x0, y0);
136  _swap_int16_t(x1, y1);
137  }
138 
139  if (x0 > x1) {
140  _swap_int16_t(x0, x1);
141  _swap_int16_t(y0, y1);
142  }
143 
144  int16_t dx, dy;
145  dx = x1 - x0;
146  dy = abs(y1 - y0);
147 
148  int16_t err = dx / 2;
149  int16_t ystep;
150 
151  if (y0 < y1) {
152  ystep = 1;
153  } else {
154  ystep = -1;
155  }
156 
157  for (; x0<=x1; x0++) {
158  if (steep) {
159  writePixel(y0, x0, color);
160  } else {
161  writePixel(x0, y0, color);
162  }
163  err -= dy;
164  if (err < 0) {
165  y0 += ystep;
166  err += dx;
167  }
168  }
169 }

349 #endif

459 void Adafruit_GFX::fillCircleHelper(int16_t x0, int16_t y0, int16_t r,
460  uint8_t corners, int16_t delta, uint16_t color) {
461 
462  int16_t f = 1 - r;
463  int16_t ddF_x = 1;
464  int16_t ddF_y = -2 * r;
465  int16_t x = 0;
466  int16_t y = r;
467  int16_t px = x;
468  int16_t py = y;
469 
470  delta++; // Avoid some +1's in the loop
471 
472  while(x < y) {
473  if (f >= 0) {
474  y--;
475  ddF_y += 2;
476  f += ddF_y;
477  }
478  x++;
479  ddF_x += 2;
480  f += ddF_x;
481  // These checks avoid double-drawing certain lines, important
482  // for the SSD1306 library which has an INVERT drawing mode.
483  if(x < (y + 1)) {
484  if(corners & 1) writeFastVLine(x0+x, y0-y, 2*y+delta, color);
485  if(corners & 2) writeFastVLine(x0-x, y0-y, 2*y+delta, color);
486  }
487  if(y != py) {
488  if(corners & 1) writeFastVLine(x0+py, y0-px, 2*px+delta, color);
489  if(corners & 2) writeFastVLine(x0-py, y0-px, 2*px+delta, color);
490  py = y;
491  }
492  px = x;
493  }
494 }

599 /**************************************************************************/
600 void Adafruit_GFX::fillTriangle(int16_t x0, int16_t y0,
601  int16_t x1, int16_t y1, int16_t x2, int16_t y2, uint16_t color) {
602 
603  int16_t a, b, y, last;
604 
605  // Sort coordinates by Y order (y2 >= y1 >= y0)
606  if (y0 > y1) {
607  _swap_int16_t(y0, y1); _swap_int16_t(x0, x1);
608  }
609  if (y1 > y2) {
610  _swap_int16_t(y2, y1); _swap_int16_t(x2, x1);
611  }
612  if (y0 > y1) {
613  _swap_int16_t(y0, y1); _swap_int16_t(x0, x1);
614  }
615 
616  startWrite();
617  if(y0 == y2) { // Handle awkward all-on-same-line case as its own thing
618  a = b = x0;
619  if(x1 < a) a = x1;
620  else if(x1 > b) b = x1;
621  if(x2 < a) a = x2;
622  else if(x2 > b) b = x2;
623  writeFastHLine(a, y0, b-a+1, color);
624  endWrite();
625  return;
626  }
627 
628  int16_t
629  dx01 = x1 - x0,
630  dy01 = y1 - y0,
631  dx02 = x2 - x0,
632  dy02 = y2 - y0,
633  dx12 = x2 - x1,
634  dy12 = y2 - y1;
635  int32_t
636  sa = 0,
637  sb = 0;
638 
639  // For upper part of triangle, find scanline crossings for segments
640  // 0-1 and 0-2. If y1=y2 (flat-bottomed triangle), the scanline y1
641  // is included here (and second loop will be skipped, avoiding a /0
642  // error there), otherwise scanline y1 is skipped here and handled
643  // in the second loop...which also avoids a /0 error here if y0=y1
644  // (flat-topped triangle).
645  if(y1 == y2) last = y1; // Include y1 scanline
646  else last = y1-1; // Skip it
647 
648  for(y=y0; y<=last; y++) {
649  a = x0 + sa / dy01;
650  b = x0 + sb / dy02;
651  sa += dx01;
652  sb += dx02;
653  /* longhand:
654  a = x0 + (x1 - x0) * (y - y0) / (y1 - y0);
655  b = x0 + (x2 - x0) * (y - y0) / (y2 - y0);
656  */
657  if(a > b) _swap_int16_t(a,b);
658  writeFastHLine(a, y, b-a+1, color);
659  }
660 
661  // For lower part of triangle, find scanline crossings for segments
662  // 0-2 and 1-2. This loop is skipped if y1=y2.
663  sa = (int32_t)dx12 * (y - y1);
664  sb = (int32_t)dx02 * (y - y0);
665  for(; y<=y2; y++) {
666  a = x1 + sa / dy12;
667  b = x0 + sb / dy02;
668  sa += dx12;
669  sb += dx02;
670  /* longhand:
671  a = x1 + (x2 - x1) * (y - y1) / (y2 - y1);
672  b = x0 + (x2 - x0) * (y - y0) / (y2 - y0);
673  */
674  if(a > b) _swap_int16_t(a,b);
675  writeFastHLine(a, y, b-a+1, color);
676  }
677  endWrite();
678 }

1063 /**************************************************************************/
1074 /**************************************************************************/
1075 void Adafruit_GFX::drawChar(int16_t x, int16_t y, unsigned char c,
1076  uint16_t color, uint16_t bg, uint8_t size_x, uint8_t size_y) {
1077 
1078  if(!gfxFont) { // 'Classic' built-in font
1079 
1080  if((x >= _width) || // Clip right
1081  (y >= _height) || // Clip bottom
1082  ((x + 6 * size_x - 1) < 0) || // Clip left
1083  ((y + 8 * size_y - 1) < 0)) // Clip top
1084  return;
1085 
1086  if(!_cp437 && (c >= 176)) c++; // Handle 'classic' charset behavior
1087 
1088  startWrite();
1089  for(int8_t i=0; i<5; i++ ) { // Char bitmap = 5 columns
1090  uint8_t line = pgm_read_byte(&font[c * 5 + i]);
1091  for(int8_t j=0; j<8; j++, line >>= 1) {
1092  if(line & 1) {
1093  if(size_x == 1 && size_y == 1)
1094  writePixel(x+i, y+j, color);
1095  else
1096  writeFillRect(x+i*size_x, y+j*size_y, size_x, size_y, color);
1097  } else if(bg != color) {
1098  if(size_x == 1 && size_y == 1)
1099  writePixel(x+i, y+j, bg);
1100  else
1101  writeFillRect(x+i*size_x, y+j*size_y, size_x, size_y, bg);
1102  }
1103  }
1104  }
1105  if(bg != color) { // If opaque, draw vertical line for last column
1106  if(size_x == 1 && size_y == 1) writeFastVLine(x+5, y, 8, bg);
1107  else writeFillRect(x+5*size_x, y, size_x, 8*size_y, bg);
1108  }
1109  endWrite();
1110 
1111  } else { // Custom font
1112 
1113  // Character is assumed previously filtered by write() to eliminate
1114  // newlines, returns, non-printable characters, etc. Calling
1115  // drawChar() directly with 'bad' characters of font may cause mayhem!
1116 
1117  c -= (uint8_t)pgm_read_byte(&gfxFont->first);
1118  GFXglyph *glyph = pgm_read_glyph_ptr(gfxFont, c);
1119  uint8_t *bitmap = pgm_read_bitmap_ptr(gfxFont);
1120 
1121  uint16_t bo = pgm_read_word(&glyph->bitmapOffset);
1122  uint8_t w = pgm_read_byte(&glyph->width),
1123  h = pgm_read_byte(&glyph->height);
1124  int8_t xo = pgm_read_byte(&glyph->xOffset),
1125  yo = pgm_read_byte(&glyph->yOffset);
1126  uint8_t xx, yy, bits = 0, bit = 0;
1127  int16_t xo16 = 0, yo16 = 0;
1128 
1129  if(size_x > 1 || size_y > 1) {
1130  xo16 = xo;
1131  yo16 = yo;
1132  }
1133 
1134  // Todo: Add character clipping here
1135 
1136  // NOTE: THERE IS NO 'BACKGROUND' COLOR OPTION ON CUSTOM FONTS.
1137  // THIS IS ON PURPOSE AND BY DESIGN. The background color feature
1138  // has typically been used with the 'classic' font to overwrite old
1139  // screen contents with new data. This ONLY works because the
1140  // characters are a uniform size; it's not a sensible thing to do with
1141  // proportionally-spaced fonts with glyphs of varying sizes (and that
1142  // may overlap). To replace previously-drawn text when using a custom
1143  // font, use the getTextBounds() function to determine the smallest
1144  // rectangle encompassing a string, erase the area with fillRect(),
1145  // then draw new text. This WILL infortunately 'blink' the text, but
1146  // is unavoidable. Drawing 'background' pixels will NOT fix this,
1147  // only creates a new set of problems. Have an idea to work around
1148  // this (a canvas object type for MCUs that can afford the RAM and
1149  // displays supporting setAddrWindow() and pushColors()), but haven't
1150  // implemented this yet.
1151 
1152  startWrite();
1153  for(yy=0; yy<h; yy++) {
1154  for(xx=0; xx<w; xx++) {
1155  if(!(bit++ & 7)) {
1156  bits = pgm_read_byte(&bitmap[bo++]);
1157  }
1158  if(bits & 0x80) {
1159  if(size_x == 1 && size_y == 1) {
1160  writePixel(x+xo+xx, y+yo+yy, color);
1161  } else {
1162  writeFillRect(x+(xo16+xx)*size_x, y+(yo16+yy)*size_y,
1163  size_x, size_y, color);
1164  }
1165  }
1166  bits <<= 1;
1167  }
1168  }
1169  endWrite();
1170 
1171  } // End classic vs custom font
1172 }
1173 /**************************************************************************/
1178 /**************************************************************************/
1179 size_t Adafruit_GFX::write(uint8_t c) {
1180  if(!gfxFont) { // 'Classic' built-in font
1181 
1182  if(c == '\n') { // Newline?
1183  cursor_x = 0; // Reset x to zero,
1184  cursor_y += textsize_y * 8; // advance y one line
1185  } else if(c != '\r') { // Ignore carriage returns
1186  if(wrap && ((cursor_x + textsize_x * 6) > _width)) { // Off right?
1187  cursor_x = 0; // Reset x to zero,
1188  cursor_y += textsize_y * 8; // advance y one line
1189  }
1190  drawChar(cursor_x, cursor_y, c, textcolor, textbgcolor, textsize_x, textsize_y);
1191  cursor_x += textsize_x * 6; // Advance x one char
1192  }
1193 
1194  } else { // Custom font
1195 
1196  if(c == '\n') {
1197  cursor_x = 0;
1198  cursor_y += (int16_t)textsize_y *
1199  (uint8_t)pgm_read_byte(&gfxFont->yAdvance);
1200  } else if(c != '\r') {
1201  uint8_t first = pgm_read_byte(&gfxFont->first);
1202  if((c >= first) && (c <= (uint8_t)pgm_read_byte(&gfxFont->last))) {
1203  GFXglyph *glyph = pgm_read_glyph_ptr(gfxFont, c - first);
1204  uint8_t w = pgm_read_byte(&glyph->width),
1205  h = pgm_read_byte(&glyph->height);
1206  if((w > 0) && (h > 0)) { // Is there an associated bitmap?
1207  int16_t xo = (int8_t)pgm_read_byte(&glyph->xOffset); // sic
1208  if(wrap && ((cursor_x + textsize_x * (xo + w)) > _width)) {
1209  cursor_x = 0;
1210  cursor_y += (int16_t)textsize_y *
1211  (uint8_t)pgm_read_byte(&gfxFont->yAdvance);
1212  }
1213  drawChar(cursor_x, cursor_y, c, textcolor, textbgcolor, textsize_x, textsize_y);
1214  }
1215  cursor_x += (uint8_t)pgm_read_byte(&glyph->xAdvance) * (int16_t)textsize_x;
1216  }
1217  }
1218 
1219  }
1220  return 1;
1221 }

1301 /**************************************************************************/
1302 void Adafruit_GFX::charBounds(char c, int16_t *x, int16_t *y,
1303  int16_t *minx, int16_t *miny, int16_t *maxx, int16_t *maxy) {
1304 
1305  if(gfxFont) {
1306 
1307  if(c == '\n') { // Newline?
1308  *x = 0; // Reset x to zero, advance y by one line
1309  *y += textsize_y * (uint8_t)pgm_read_byte(&gfxFont->yAdvance);
1310  } else if(c != '\r') { // Not a carriage return; is normal char
1311  uint8_t first = pgm_read_byte(&gfxFont->first),
1312  last = pgm_read_byte(&gfxFont->last);
1313  if((c >= first) && (c <= last)) { // Char present in this font?
1314  GFXglyph *glyph = pgm_read_glyph_ptr(gfxFont, c - first);
1315  uint8_t gw = pgm_read_byte(&glyph->width),
1316  gh = pgm_read_byte(&glyph->height),
1317  xa = pgm_read_byte(&glyph->xAdvance);
1318  int8_t xo = pgm_read_byte(&glyph->xOffset),
1319  yo = pgm_read_byte(&glyph->yOffset);
1320  if(wrap && ((*x+(((int16_t)xo+gw)*textsize_x)) > _width)) {
1321  *x = 0; // Reset x to zero, advance y by one line
1322  *y += textsize_y * (uint8_t)pgm_read_byte(&gfxFont->yAdvance);
1323  }
1324  int16_t tsx = (int16_t)textsize_x,
1325  tsy = (int16_t)textsize_y,
1326  x1 = *x + xo * tsx,
1327  y1 = *y + yo * tsy,
1328  x2 = x1 + gw * tsx - 1,
1329  y2 = y1 + gh * tsy - 1;
1330  if(x1 < *minx) *minx = x1;
1331  if(y1 < *miny) *miny = y1;
1332  if(x2 > *maxx) *maxx = x2;
1333  if(y2 > *maxy) *maxy = y2;
1334  *x += xa * tsx;
1335  }
1336  }
1337 
1338  } else { // Default font
1339 
1340  if(c == '\n') { // Newline?
1341  *x = 0; // Reset x to zero,
1342  *y += textsize_y * 8; // advance y one line
1343  // min/max x/y unchaged -- that waits for next 'normal' character
1344  } else if(c != '\r') { // Normal char; ignore carriage returns
1345  if(wrap && ((*x + textsize_x * 6) > _width)) { // Off right?
1346  *x = 0; // Reset x to zero,
1347  *y += textsize_y * 8; // advance y one line
1348  }
1349  int x2 = *x + textsize_x * 6 - 1, // Lower-right pixel of char
1350  y2 = *y + textsize_y * 8 - 1;
1351  if(x2 > *maxx) *maxx = x2; // Track max x, y
1352  if(y2 > *maxy) *maxy = y2;
1353  if(*x < *minx) *minx = *x; // Track min x, y
1354  if(*y < *miny) *miny = *y;
1355  *x += textsize_x * 6; // Advance x one char
1356  }
1357  }
1358 }
