94 /* Find variable in Object Dictionary *****************************************/
95 static void findVariable(CO_trace_t *trace) {
96  bool_t err = false;
97  uint16_t index;
98  uint8_t subIndex;
99  uint8_t dataLen;
100  void *OdDataPtr = NULL;
101  unsigned dtIndex = 0;
102 
103  /* parse mapping */
104  index = (uint16_t) ((*trace->map) >> 16);
105  subIndex = (uint8_t) ((*trace->map) >> 8);
106  dataLen = (uint8_t) (*trace->map);
107  if((dataLen & 0x07) != 0) { /* data length must be byte aligned */
108  err = true;
109  }
110  dataLen >>= 3; /* in bytes now */
111  if(dataLen == 0) {
112  dataLen = 4;
113  }
114 
115  /* find mapped variable, if map available */
116  if(!err && (index != 0 || subIndex != 0)) {
117  uint16_t entryNo = CO_OD_find(trace->SDO, index);
118 
119  if(index >= 0x1000 && entryNo != 0xFFFF && subIndex <= trace->SDO->OD[entryNo].maxSubIndex) {
120  OdDataPtr = CO_OD_getDataPointer(trace->SDO, entryNo, subIndex);
121  }
122 
123  if(OdDataPtr != NULL) {
124  uint16_t len = CO_OD_getLength(trace->SDO, entryNo, subIndex);
125 
126  if(len < dataLen) {
127  dataLen = len;
128  }
129  }
130  else {
131  err = true;
132  }
133  }
134 
135  /* Get function pointers for correct data type */
136  if(!err) {
137  /* first sequence: data length */
138  switch(dataLen) {
139  case 1: dtIndex = 0; break;
140  case 2: dtIndex = 1; break;
141  case 4: dtIndex = 2; break;
142  default: err = true; break;
143  }
144  /* second sequence: signed or unsigned */
145  if(((*trace->format) & 1) == 1) {
146  dtIndex += 3;
147  }
148  /* third sequence: Output type */
149  dtIndex += ((*trace->format) >> 1) * 6;
150 
151  if(dtIndex > (sizeof(dataTypes) / sizeof(CO_trace_dataType_t))) {
152  err = true;
153  }
154  }
155 
156  /* set output variables */
157  if(!err) {
158  if(OdDataPtr != NULL) {
159  trace->OD_variable = OdDataPtr;
160  }
161  else {
162  trace->OD_variable = trace->value;
163  }
164  trace->dt = &dataTypes[dtIndex];
165  }
166  else {
167  trace->OD_variable = NULL;
168  trace->dt = NULL;
169  }
170 }

173 /* OD function for accessing _OD_traceConfig_ (index 0x2300+) from SDO server.
174  * For more information see file CO_SDO.h. */
175 static CO_SDO_abortCode_t CO_ODF_traceConfig(CO_ODF_arg_t *ODF_arg) {
176  CO_trace_t *trace;
177  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
178 
179  trace = (CO_trace_t*) ODF_arg->object;
180 
181  switch(ODF_arg->subIndex) {
182  case 1: /* size */
183  if(ODF_arg->reading) {
184  uint32_t *value = (uint32_t*) ODF_arg->data;
185  *value = trace->bufferSize;
186  }
187  break;
188 
189  case 2: /* axisNo (trace enabled if nonzero) */
190  if(ODF_arg->reading) {
191  uint8_t *value = (uint8_t*) ODF_arg->data;
192  if(!trace->enabled) {
193  *value = 0;
194  }
195  }
196  else {
197  uint8_t *value = (uint8_t*) ODF_arg->data;
198 
199  if(*value == 0) {
200  trace->enabled = false;
201  }
202  else if(!trace->enabled) {
203  if(trace->bufferSize == 0) {
204  ret = CO_SDO_AB_OUT_OF_MEM;
205  }
206  else {
207  /* set trace->OD_variable and trace->dt, based on 'map' and 'format' */
208  findVariable(trace);
209 
210  if(trace->OD_variable != NULL) {
211  *trace->value = 0;
212  *trace->minValue = 0;
213  *trace->maxValue = 0;
214  *trace->triggerTime = 0;
215  trace->valuePrev = 0;
216  trace->readPtr = 0;
217  trace->writePtr = 0;
218  trace->enabled = true;
219  }
220  else {
221  ret = CO_SDO_AB_NO_MAP;
222  }
223  }
224  }
225  }
226  break;
227 
228  case 5: /* map */
229  case 6: /* format */
230  if(!ODF_arg->reading) {
231  if(trace->enabled) {
232  ret = CO_SDO_AB_INVALID_VALUE;
233  }
234  }
235  break;
236  }
237 
238  return ret;
239 }

242 /* OD function for accessing _OD_trace_ (index 0x2400+) from SDO server.
243  * For more information see file CO_SDO.h. */
244 static CO_SDO_abortCode_t CO_ODF_trace(CO_ODF_arg_t *ODF_arg) {
245  CO_trace_t *trace;
246  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
247 
248  trace = (CO_trace_t*) ODF_arg->object;
249 
250  switch(ODF_arg->subIndex) {
251  case 1: /* size */
252  if(ODF_arg->reading) {
253  uint32_t *value = (uint32_t*) ODF_arg->data;
254  uint32_t size = trace->bufferSize;
255  uint32_t wp = trace->writePtr;
256  uint32_t rp = trace->readPtr;
257 
258  if(wp >= rp) {
259  *value = wp - rp;
260  }
261  else {
262  *value = size - rp + wp;
263  }
264  }
265  else {
266  uint32_t *value = (uint32_t*) ODF_arg->data;
267 
268  if(*value == 0) {
269  /* clear buffer, handle race conditions */
270  while(trace->readPtr != 0 || trace->writePtr != 0) {
271  trace->readPtr = 0;
272  trace->writePtr = 0;
273  *trace->triggerTime = 0;
274  }
275  }
276  else {
277  ret = CO_SDO_AB_INVALID_VALUE;
278  }
279  }
280  break;
281 
282  case 5: /* plot */
283  if(ODF_arg->reading) {
284  /* This plot will be transmitted as domain data type. String data
285  * will be printed directly to SDO buffer. If there is more data
286  * to print, than is the size of SDO buffer, then this function
287  * will be called multiple times until internal trace buffer is
288  * empty. Internal trace buffer is circular buffer. It is accessed
289  * by this function and by higher priority thread. If this buffer
290  * is full, there is a danger for race condition. First records
291  * from trace buffer may be overwritten somewhere between. If this
292  * is detected, then do{}while() loop tries printing again. */
293  if(trace->bufferSize == 0 || ODF_arg->dataLength < 100) {
294  ret = CO_SDO_AB_OUT_OF_MEM;
295  }
296  else if(trace->readPtr == trace->writePtr) {
297  ret = CO_SDO_AB_NO_DATA;
298  }
299  else {
300  uint32_t rp, t, v, len, freeLen;
301  char *s;
302  bool_t readPtrOverflowed; /* for handling race conditions */
303 
304  /* repeat everything, if trace->readPtr was overflowed in CO_trace_process */
305  do {
306  readPtrOverflowed = false;
307  s = (char*) ODF_arg->data;
308  freeLen = ODF_arg->dataLength;
309 
310  rp = trace->readPtr;
311 
312  /* start plot, increment variables, verify overflow */
313  if(ODF_arg->firstSegment) {
314  t = trace->timeBuffer[rp];
315  v = trace->valueBuffer[rp];
316  rp ++;
317  if(++trace->readPtr == trace->bufferSize) {
318  trace->readPtr = 0;
319  if(rp != trace->bufferSize) {
320  readPtrOverflowed = true;
321  continue;
322  }
323  rp = 0;
324  }
325  if(rp != trace->readPtr) {
326  readPtrOverflowed = true;
327  continue;
328  }
329  len = trace->dt->printPointStart(s, freeLen, t, v);
330  s += len;
331  freeLen -= len;
332  }
333 
334  /* print other points */
335  if(rp != trace->writePtr) {
336  for(;;) {
337  t = trace->timeBuffer[rp];
338  v = trace->valueBuffer[rp];
339  rp ++;
340  if(++trace->readPtr == trace->bufferSize) {
341  trace->readPtr = 0;
342  if(rp != trace->bufferSize && ODF_arg->firstSegment) {
343  readPtrOverflowed = true;
344  break;
345  }
346  rp = 0;
347  }
348  if(rp != trace->readPtr && ODF_arg->firstSegment) {
349  readPtrOverflowed = true;
350  break;
351  }
352 
353  /* If internal buffer is empty, end transfer */
354  if(rp == trace->writePtr) {
355  /* If there is last time stamp, point will be printed at the end */
356  if(t != trace->lastTimeStamp) {
357  len = trace->dt->printPoint(s, freeLen, t, v);
358  s += len;
359  freeLen -= len;
360  }
361  ODF_arg->lastSegment = true;
362  break;
363  }
364  len = trace->dt->printPoint(s, freeLen, t, v);
365  s += len;
366  freeLen -= len;
367 
368  /* if output buffer is full, next data will be sent later */
369  if(freeLen < 50) {
370  ODF_arg->lastSegment = false;
371  break;
372  }
373  }
374  }
375 
376  /* print last point */
377  if(!readPtrOverflowed && ODF_arg->lastSegment) {
378  v = trace->valuePrev;
379  t = trace->lastTimeStamp;
380  len = trace->dt->printPointEnd(s, freeLen, t, v);
381  s += len;
382  freeLen -= len;
383  }
384  } while(readPtrOverflowed);
385 
386  ODF_arg->dataLength -= freeLen;
387  }
388  }
389  break;
390  }
391 
392  return ret;
393 }

453 /******************************************************************************/
454 void CO_trace_process(CO_trace_t *trace, uint32_t timestamp) {
455  if(trace->enabled) {
456 
457  int32_t val = trace->dt->pGetValue(trace->OD_variable);
458 
459  if(val != trace->valuePrev) {
460  /* Verify, if value passed threshold */
461  if((*trace->trigger & 1) != 0 && trace->valuePrev < *trace->threshold && val >= *trace->threshold) {
462  *trace->triggerTime = timestamp;
463  }
464  if((*trace->trigger & 2) != 0 && trace->valuePrev < *trace->threshold && val >= *trace->threshold) {
465  *trace->triggerTime = timestamp;
466  }
467 
468  /* Write value and verify min/max */
469  if(trace->value != trace->OD_variable) {
470  *trace->value = val;
471  }
472  trace->valuePrev = val;
473  if(*trace->minValue > val) {
474  *trace->minValue = val;
475  }
476  if(*trace->maxValue < val) {
477  *trace->maxValue = val;
478  }
479 
480  /* write buffers and update pointers */
481  trace->timeBuffer[trace->writePtr] = timestamp;
482  trace->valueBuffer[trace->writePtr] = val;
483  if(++trace->writePtr == trace->bufferSize) {
484  trace->writePtr = 0;
485  }
486  if(trace->writePtr == trace->readPtr) {
487  if(++trace->readPtr == trace->bufferSize) {
488  trace->readPtr = 0;
489  }
490  }
491  }
492  else {
493  /* if buffer is empty, make first record */
494  if(trace->writePtr == trace->readPtr) {
495  /* write buffers and update pointers */
496  trace->timeBuffer[trace->writePtr] = timestamp;
497  trace->valueBuffer[trace->writePtr] = val;
498  if(++trace->writePtr == trace->bufferSize) {
499  trace->writePtr = 0;
500  }
501  }
502  }
503  trace->lastTimeStamp = timestamp;
504  }
505 }
