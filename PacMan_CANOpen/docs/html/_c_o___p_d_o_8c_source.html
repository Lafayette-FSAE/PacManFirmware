34 /*
35  * Read received message from CAN module.
36  *
37  * Function will be called (by CAN receive interrupt) every time, when CAN
38  * message with correct identifier will be received. For more information and
39  * description of parameters see file CO_driver.h.
40  * If new message arrives and previous message wasn't processed yet, then
41  * previous message will be lost and overwritten by new message. That's OK with PDOs.
42  */
43 static void CO_PDO_receive(void *object, const CO_CANrxMsg_t *msg){
44  CO_RPDO_t *RPDO;
45 
46  RPDO = (CO_RPDO_t*)object; /* this is the correct pointer type of the first argument */
47 
48  if( (RPDO->valid) &&
49  (*RPDO->operatingState == CO_NMT_OPERATIONAL) &&
50  (msg->DLC >= RPDO->dataLength))
51  {
52  if(RPDO->SYNC && RPDO->synchronous && RPDO->SYNC->CANrxToggle) {
53  /* copy data into second buffer and set 'new message' flag */
54  RPDO->CANrxData[1][0] = msg->data[0];
55  RPDO->CANrxData[1][1] = msg->data[1];
56  RPDO->CANrxData[1][2] = msg->data[2];
57  RPDO->CANrxData[1][3] = msg->data[3];
58  RPDO->CANrxData[1][4] = msg->data[4];
59  RPDO->CANrxData[1][5] = msg->data[5];
60  RPDO->CANrxData[1][6] = msg->data[6];
61  RPDO->CANrxData[1][7] = msg->data[7];
62 
63  SET_CANrxNew(RPDO->CANrxNew[1]);
64  }
65  else {
66  /* copy data into default buffer and set 'new message' flag */
67  RPDO->CANrxData[0][0] = msg->data[0];
68  RPDO->CANrxData[0][1] = msg->data[1];
69  RPDO->CANrxData[0][2] = msg->data[2];
70  RPDO->CANrxData[0][3] = msg->data[3];
71  RPDO->CANrxData[0][4] = msg->data[4];
72  RPDO->CANrxData[0][5] = msg->data[5];
73  RPDO->CANrxData[0][6] = msg->data[6];
74  RPDO->CANrxData[0][7] = msg->data[7];
75 
76  SET_CANrxNew(RPDO->CANrxNew[0]);
77  }
78  }
79 }

82 /*
83  * Configure RPDO Communication parameter.
84  *
85  * Function is called from commuincation reset or when parameter changes.
86  *
87  * Function configures following variable from CO_RPDO_t: _valid_. It also
88  * configures CAN rx buffer. If configuration fails, emergency message is send
89  * and device is not able to enter NMT operational.
90  *
91  * @param RPDO RPDO object.
92  * @param COB_IDUsedByRPDO _RPDO communication parameter_, _COB-ID for PDO_ variable
93  * from Object dictionary (index 0x1400+, subindex 1).
94  */
95 static void CO_RPDOconfigCom(CO_RPDO_t* RPDO, uint32_t COB_IDUsedByRPDO){
96  uint16_t ID;
97  CO_ReturnError_t r;
98 
99  ID = (uint16_t)COB_IDUsedByRPDO;
100 
101  /* is RPDO used? */
102  if((COB_IDUsedByRPDO & 0xBFFFF800L) == 0 && RPDO->dataLength && ID){
103  /* is used default COB-ID? */
104  if(ID == RPDO->defaultCOB_ID) ID += RPDO->nodeId;
105  RPDO->valid = true;
106  RPDO->synchronous = (RPDO->RPDOCommPar->transmissionType <= 240) ? true : false;
107  }
108  else{
109  ID = 0;
110  RPDO->valid = false;
111  CLEAR_CANrxNew(RPDO->CANrxNew[0]);
112  CLEAR_CANrxNew(RPDO->CANrxNew[1]);
113  }
114  printf("\nPDO Print: %u ", RPDO->CANdevRxIdx);
115  r = CO_CANrxBufferInit(
116  RPDO->CANdevRx, /* CAN device */
117  RPDO->CANdevRxIdx, /* rx buffer index */
118  ID, /* CAN identifier */
119  0x7FF, /* mask */
120  0, /* rtr */
121  (void*)RPDO, /* object passed to receive function */
122  CO_PDO_receive); /* this function will process received message */
123  if(r != CO_ERROR_NO){
124  RPDO->valid = false;
125  CLEAR_CANrxNew(RPDO->CANrxNew[0]);
126  CLEAR_CANrxNew(RPDO->CANrxNew[1]);
127  }
128 }

175 /*
176  * Find mapped variable in Object Dictionary.
177  *
178  * Function is called from CO_R(T)PDOconfigMap or when mapping parameter changes.
179  *
180  * @param SDO SDO object.
181  * @param map PDO mapping parameter.
182  * @param R_T 0 for RPDO map, 1 for TPDO map.
183  * @param ppData Pointer to returning parameter: pointer to data of mapped variable.
184  * @param pLength Pointer to returning parameter: *add* length of mapped variable.
185  * @param pSendIfCOSFlags Pointer to returning parameter: sendIfCOSFlags variable.
186  * @param pIsMultibyteVar Pointer to returning parameter: true for multibyte variable.
187  *
188  * @return 0 on success, otherwise SDO abort code.
189  */
190 static uint32_t CO_PDOfindMap(
191  CO_SDO_t *SDO,
192  uint32_t map,
193  uint8_t R_T,
194  uint8_t **ppData,
195  uint8_t *pLength,
196  uint8_t *pSendIfCOSFlags,
197  uint8_t *pIsMultibyteVar)
198 {
199  uint16_t entryNo;
200  uint16_t index;
201  uint8_t subIndex;
202  uint8_t dataLen;
203  uint8_t objectLen;
204  uint8_t attr;
205 
206  index = (uint16_t)(map>>16);
207  subIndex = (uint8_t)(map>>8);
208  dataLen = (uint8_t) map; /* data length in bits */
209 
210  /* data length must be byte aligned */
211  if(dataLen&0x07) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
212 
213  dataLen >>= 3; /* new data length is in bytes */
214  *pLength += dataLen;
215 
216  /* total PDO length can not be more than 8 bytes */
217  if(*pLength > 8) return CO_SDO_AB_MAP_LEN; /* The number and length of the objects to be mapped would exceed PDO length. */
218 
219  /* is there a reference to dummy entries */
220  if(index <=7 && subIndex == 0){
221  static uint32_t dummyTX = 0;
222  static uint32_t dummyRX;
223  uint8_t dummySize = 4;
224 
225  if(index<2) dummySize = 0;
226  else if(index==2 || index==5) dummySize = 1;
227  else if(index==3 || index==6) dummySize = 2;
228 
229  /* is size of variable big enough for map */
230  if(dummySize < dataLen) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
231 
232  /* Data and ODE pointer */
233  if(R_T == 0) *ppData = (uint8_t*) &dummyRX;
234  else *ppData = (uint8_t*) &dummyTX;
235 
236  return 0;
237  }
238 
239  /* find object in Object Dictionary */
240  entryNo = CO_OD_find(SDO, index);
241 
242  /* Does object exist in OD? */
243  if(entryNo == 0xFFFF || subIndex > SDO->OD[entryNo].maxSubIndex)
244  return CO_SDO_AB_NOT_EXIST; /* Object does not exist in the object dictionary. */
245 
246  attr = CO_OD_getAttribute(SDO, entryNo, subIndex);
247  /* Is object Mappable for RPDO? */
248  if(R_T==0 && !((attr&CO_ODA_RPDO_MAPABLE) && (attr&CO_ODA_WRITEABLE))) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
249  /* Is object Mappable for TPDO? */
250  if(R_T!=0 && !((attr&CO_ODA_TPDO_MAPABLE) && (attr&CO_ODA_READABLE))) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
251 
252  /* is size of variable big enough for map */
253  objectLen = CO_OD_getLength(SDO, entryNo, subIndex);
254  if(objectLen < dataLen) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
255 
256  /* mark multibyte variable */
257  *pIsMultibyteVar = (attr&CO_ODA_MB_VALUE) ? 1 : 0;
258 
259  /* pointer to data */
260  *ppData = (uint8_t*) CO_OD_getDataPointer(SDO, entryNo, subIndex);
261 #ifdef CO_BIG_ENDIAN
262  /* skip unused MSB bytes */
263  if(*pIsMultibyteVar){
264  *ppData += objectLen - dataLen;
265  }
266 #endif
267 
268  /* setup change of state flags */
269  if(attr&CO_ODA_TPDO_DETECT_COS){
270  int16_t i;
271  for(i=*pLength-dataLen; i<*pLength; i++){
272  *pSendIfCOSFlags |= 1<<i;
273  }
274  }
275 
276  return 0;
277 }

280 /*
281  * Configure RPDO Mapping parameter.
282  *
283  * Function is called from communication reset or when parameter changes.
284  *
285  * Function configures following variables from CO_RPDO_t: _dataLength_ and
286  * _mapPointer_.
287  *
288  * @param RPDO RPDO object.
289  * @param noOfMappedObjects Number of mapped object (from OD).
290  *
291  * @return 0 on success, otherwise SDO abort code.
292  */
293 static uint32_t CO_RPDOconfigMap(CO_RPDO_t* RPDO, uint8_t noOfMappedObjects){
294  int16_t i;
295  uint8_t length = 0;
296  uint32_t ret = 0;
297  const uint32_t* pMap = &RPDO->RPDOMapPar->mappedObject1;
298 
299  for(i=noOfMappedObjects; i>0; i--){
300  int16_t j;
301  uint8_t* pData;
302  uint8_t dummy = 0;
303  uint8_t prevLength = length;
304  uint8_t MBvar;
305  uint32_t map = *(pMap++);
306 
307  /* function do much checking of errors in map */
308  ret = CO_PDOfindMap(
309  RPDO->SDO,
310  map,
311  0,
312  &pData,
313  &length,
314  &dummy,
315  &MBvar);
316  if(ret){
317  length = 0;
318  CO_errorReport(RPDO->em, CO_EM_PDO_WRONG_MAPPING, CO_EMC_PROTOCOL_ERROR, map);
319  break;
320  }
321 
322  /* write PDO data pointers */
323 #ifdef CO_BIG_ENDIAN
324  if(MBvar){
325  for(j=length-1; j>=prevLength; j--)
326  RPDO->mapPointer[j] = pData++;
327  }
328  else{
329  for(j=prevLength; j<length; j++)
330  RPDO->mapPointer[j] = pData++;
331  }
332 #else
333  for(j=prevLength; j<length; j++){
334  RPDO->mapPointer[j] = pData++;
335  }
336 #endif
337 
338  }
339 
340  RPDO->dataLength = length;
341 
342  return ret;
343 }

413 /*
414  * Function for accessing _RPDO communication parameter_ (index 0x1400+) from SDO server.
415  *
416  * For more information see file CO_SDO.h.
417  */
418 static CO_SDO_abortCode_t CO_ODF_RPDOcom(CO_ODF_arg_t *ODF_arg){
419  CO_RPDO_t *RPDO;
420 
421  RPDO = (CO_RPDO_t*) ODF_arg->object;
422 
423  /* Reading Object Dictionary variable */
424  if(ODF_arg->reading){
425  if(ODF_arg->subIndex == 1){
426  uint32_t value = CO_getUint32(ODF_arg->data);
427 
428  /* if default COB ID is used, write default value here */
429  if(((value)&0xFFFF) == RPDO->defaultCOB_ID && RPDO->defaultCOB_ID)
430  value += RPDO->nodeId;
431 
432  /* If PDO is not valid, set bit 31 */
433  if(!RPDO->valid) value |= 0x80000000L;
434 
435  CO_setUint32(ODF_arg->data, value);
436  }
437  return CO_SDO_AB_NONE;
438  }
439 
440  /* Writing Object Dictionary variable */
441  if(RPDO->restrictionFlags & 0x04)
442  return CO_SDO_AB_READONLY; /* Attempt to write a read only object. */
443  if(*RPDO->operatingState == CO_NMT_OPERATIONAL && (RPDO->restrictionFlags & 0x01))
444  return CO_SDO_AB_DATA_DEV_STATE; /* Data cannot be transferred or stored to the application because of the present device state. */
445 
446  if(ODF_arg->subIndex == 1){ /* COB_ID */
447  uint32_t value = CO_getUint32(ODF_arg->data);
448 
449  /* bits 11...29 must be zero */
450  if(value & 0x3FFF8000L)
451  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
452 
453  /* if default COB-ID is being written, write defaultCOB_ID without nodeId */
454  if(((value)&0xFFFF) == (RPDO->defaultCOB_ID + RPDO->nodeId)){
455  value &= 0xC0000000L;
456  value += RPDO->defaultCOB_ID;
457  CO_setUint32(ODF_arg->data, value);
458  }
459 
460  /* if PDO is valid, bits 0..29 can not be changed */
461  if(RPDO->valid && ((value ^ RPDO->RPDOCommPar->COB_IDUsedByRPDO) & 0x3FFFFFFFL))
462  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
463 
464  /* configure RPDO */
465  CO_RPDOconfigCom(RPDO, value);
466  }
467  else if(ODF_arg->subIndex == 2){ /* Transmission_type */
468  uint8_t *value = (uint8_t*) ODF_arg->data;
469  bool_t synchronousPrev = RPDO->synchronous;
470 
471  /* values from 241...253 are not valid */
472  if(*value >= 241 && *value <= 253)
473  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
474 
475  RPDO->synchronous = (*value <= 240) ? true : false;
476 
477  /* Remove old message from second buffer. */
478  if(RPDO->synchronous != synchronousPrev) {
479  CLEAR_CANrxNew(RPDO->CANrxNew[1]);
480  }
481  }
482 
483  return CO_SDO_AB_NONE;
484 }

539  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
540

559  TPDO->inhibitTimer = 0;
560  }

566  else if(ODF_arg->subIndex == 6){ /* SYNC start value */
567  uint8_t *value = (uint8_t*) ODF_arg->data;
568

570  if(TPDO->valid)
571  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
572 
573  /* values from 240...255 are not valid */
574  if(*value > 240)
575  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
576  }
577 
578  return CO_SDO_AB_NONE;
579 }

971 /******************************************************************************/
972 void CO_TPDO_process(
973  CO_TPDO_t *TPDO,
974  bool_t syncWas,
975  uint32_t timeDifference_us)
976 {
977  if(TPDO->valid && *TPDO->operatingState == CO_NMT_OPERATIONAL){
978 
979  /* Send PDO by application request or by Event timer */
980  if(TPDO->TPDOCommPar->transmissionType >= 253){
981  if(TPDO->inhibitTimer == 0 && (TPDO->sendRequest || (TPDO->TPDOCommPar->eventTimer && TPDO->eventTimer == 0))){
982  if(CO_TPDOsend(TPDO) == CO_ERROR_NO){
983  /* successfully sent */
984  TPDO->inhibitTimer = ((uint32_t) TPDO->TPDOCommPar->inhibitTime) * 100;
985  TPDO->eventTimer = ((uint32_t) TPDO->TPDOCommPar->eventTimer) * 1000;
986  }
987  }
988  }
989 
990  /* Synchronous PDOs */
991  else if(TPDO->SYNC && syncWas){
992  /* send synchronous acyclic PDO */
993  if(TPDO->TPDOCommPar->transmissionType == 0){
994  if(TPDO->sendRequest) CO_TPDOsend(TPDO);
995  }
996  /* send synchronous cyclic PDO */
997  else{
998  /* is the start of synchronous TPDO transmission */
999  if(TPDO->syncCounter == 255){
1000  if(TPDO->SYNC->counterOverflowValue && TPDO->TPDOCommPar->SYNCStartValue)
1001  TPDO->syncCounter = 254; /* SYNCStartValue is in use */
1002  else
1003  TPDO->syncCounter = TPDO->TPDOCommPar->transmissionType;
1004  }
1005  /* if the SYNCStartValue is in use, start first TPDO after SYNC with matched SYNCStartValue. */
1006  if(TPDO->syncCounter == 254){
1007  if(TPDO->SYNC->counter == TPDO->TPDOCommPar->SYNCStartValue){
1008  TPDO->syncCounter = TPDO->TPDOCommPar->transmissionType;
1009  CO_TPDOsend(TPDO);
1010  }
1011  }
1012  /* Send PDO after every N-th Sync */
1013  else if(--TPDO->syncCounter == 0){
1014  TPDO->syncCounter = TPDO->TPDOCommPar->transmissionType;
1015  CO_TPDOsend(TPDO);
1016  }
1017  }
1018  }
1019 
1020  }
1021  else{
1022  /* Not operational or valid. Force TPDO first send after operational or valid. */
1023  if(TPDO->TPDOCommPar->transmissionType>=254) TPDO->sendRequest = 1;
1024  else TPDO->sendRequest = 0;
1025  }
1026 
1027  /* update timers */
1028  TPDO->inhibitTimer = (TPDO->inhibitTimer > timeDifference_us) ? (TPDO->inhibitTimer - timeDifference_us) : 0;
1029  TPDO->eventTimer = (TPDO->eventTimer > timeDifference_us) ? (TPDO->eventTimer - timeDifference_us) : 0;
1030 }
