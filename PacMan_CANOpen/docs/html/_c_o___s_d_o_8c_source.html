164 /*
165  * Read received message from CAN module.
166  *
167  * Function will be called (by CAN receive interrupt) every time, when CAN
168  * message with correct identifier will be received. For more information and
169  * description of parameters see file CO_driver.h.
170  */
171 static void CO_SDO_receive(void *object, const CO_CANrxMsg_t *msg);
172 static void CO_SDO_receive(void *object, const CO_CANrxMsg_t *msg){
173  CO_SDO_t *SDO;
174 
175  SDO = (CO_SDO_t*)object; /* this is the correct pointer type of the first argument */
176  //printf("YOure on the right track");
177 
178  /* WARNING: When doing a SDO block upload and immediately after that
179  * starting another SDO request, this request is dropped. Especially if
180  * processing function has slow response.
181  * See: https://github.com/CANopenNode/CANopenNode/issues/39 */
182 
183  /* verify message length and message overflow (previous message was not processed yet) */
184  if((msg->DLC == 8U) && (!IS_CANrxNew(SDO->CANrxNew))){
185  if(SDO->state != CO_SDO_ST_DOWNLOAD_BL_SUBBLOCK) {
186  /* copy data and set 'new message' flag */
187  SDO->CANrxData[0] = msg->data[0];
188  SDO->CANrxData[1] = msg->data[1];
189  SDO->CANrxData[2] = msg->data[2];
190  SDO->CANrxData[3] = msg->data[3];
191  SDO->CANrxData[4] = msg->data[4];
192  SDO->CANrxData[5] = msg->data[5];
193  SDO->CANrxData[6] = msg->data[6];
194  SDO->CANrxData[7] = msg->data[7];
195 
196  SET_CANrxNew(SDO->CANrxNew);
197  }
198  else {
199  /* block download, copy data directly */
200  uint8_t seqno;
201 
202  SDO->CANrxData[0] = msg->data[0];
203  seqno = SDO->CANrxData[0] & 0x7fU;
204  SDO->timeoutTimer = 0;
205 
206  /* check correct sequence number. */
207  if(seqno == (SDO->sequence + 1U)) {
208  /* sequence is correct */
209  uint8_t i;
210 
211  SDO->sequence++;
212 
213  /* copy data */
214  for(i=1; i<8; i++) {
215  SDO->ODF_arg.data[SDO->bufferOffset++] = msg->data[i]; //SDO->ODF_arg.data is equal as SDO->databuffer
216  if(SDO->bufferOffset >= CO_SDO_BUFFER_SIZE) {
217  /* buffer full, break reception */
218  SDO->state = CO_SDO_ST_DOWNLOAD_BL_SUB_RESP;
219  SET_CANrxNew(SDO->CANrxNew);
220  break;
221  }
222  }
223 
224  /* break reception if last segment or block sequence is too large */
225  if(((SDO->CANrxData[0] & 0x80U) == 0x80U) || (SDO->sequence >= SDO->blksize)) {
226  SDO->state = CO_SDO_ST_DOWNLOAD_BL_SUB_RESP;
227  SET_CANrxNew(SDO->CANrxNew);
228  }
229  }
230  else if((seqno == SDO->sequence) || (SDO->sequence == 0U)){
231  /* Ignore message, if it is duplicate or if sequence didn't started yet. */
232  }
233  else {
234  /* seqno is totally wrong, break reception. */
235  SDO->state = CO_SDO_ST_DOWNLOAD_BL_SUB_RESP;
236  SET_CANrxNew(SDO->CANrxNew);
237  }
238  }
239 
240  /* Optional signal to RTOS, which can resume task, which handles SDO server. */
241  if(IS_CANrxNew(SDO->CANrxNew) && SDO->pFunctSignal != NULL) {
242  SDO->pFunctSignal();
243  }
244  }
245 }

274  CO_SDO_t *SDO,
275  uint32_t COB_IDClientToServer,

295  uint16_t i;
296 
297  SDO->ownOD = true;

312  SDO->OD = parentSDO->OD;

314  SDO->ODExtensions = parentSDO->ODExtensions;

319  SDO->state = CO_SDO_ST_IDLE;
320  CLEAR_CANrxNew(SDO->CANrxNew);
321  SDO->pFunctSignal = NULL;
322 
323 
324  /* Configure Object dictionary entry at index 0x1200 */
325  if(ObjDictIndex_SDOServerParameter == OD_H1200_SDO_SERVER_PARAM){
326  CO_OD_configure(SDO, ObjDictIndex_SDOServerParameter, CO_ODF_1200, (void*)&SDO->nodeId, 0U, 0U);

328

345

347  SDO->CANdevTx = CANdevTx;
348  SDO->CANtxBuff = CO_CANtxBufferInit(
349  CANdevTx, /* CAN device */
350  CANdevTxIdx, /* index of specific buffer inside CAN module */
351  COB_IDServerToClient, /* CAN identifier */
352  0, /* rtr */
353  8, /* number of data bytes */
354  0); /* synchronous message flag bit */
355 
356  return CO_ERROR_NO;
357 }

403 /******************************************************************************/
404 uint16_t CO_OD_find(CO_SDO_t *SDO, uint16_t index){
405  /* Fast search in ordered Object Dictionary. If indexes are mixed, this won't work. */
406  /* If Object Dictionary has up to 2^N entries, then N is max number of loop passes. */
407  uint16_t cur, min, max;
408  const CO_OD_entry_t* object;
409 
410  min = 0U;
411  max = SDO->ODSize - 1U;
412  while(min < max){
413  cur = (min + max) / 2;
414  object = &SDO->OD[cur];
415  /* Is object matched */
416  if(index == object->index){
417  return cur;
418  }
419  if(index < object->index){
420  max = cur;
421  if(max) max--;
422  }
423  else
424  min = cur + 1U;
425  }
426 
427  if(min == max){
428  object = &SDO->OD[min];
429  /* Is object matched */
430  if(index == object->index){
431  return min;
432  }
433  }
434 
435  return 0xFFFFU; /* object does not exist in OD */
436 }

662

666  }
667  /* if domain, Object dictionary function MUST exist */
668  else{

672  }

692 
693  SDO->ODF_arg.offset += SDO->ODF_arg.dataLength;

695

706  *(buf1++) = *buf2;

710 #endif

802 /******************************************************************************/
803 int8_t CO_SDO_process(
804  CO_SDO_t *SDO,
805  bool_t NMTisPreOrOperational,
806  uint16_t timeDifference_ms,
807  uint16_t SDOtimeoutTime,
808  uint16_t *timerNext_ms)
809 {
810  CO_SDO_state_t state = CO_SDO_ST_IDLE;
811  bool_t timeoutSubblockDownolad = false;
812  bool_t sendResponse = false;
813 
814  /* return if idle */
815  if((SDO->state == CO_SDO_ST_IDLE) && (!IS_CANrxNew(SDO->CANrxNew))){
816  return 0;
817  }
818 
819  /* SDO is allowed to work only in operational or pre-operational NMT state */
820  if(!NMTisPreOrOperational){
821  SDO->state = CO_SDO_ST_IDLE;
822  CLEAR_CANrxNew(SDO->CANrxNew);
823  return 0;
824  }
825 
826  /* Is something new to process? */
827  if((!SDO->CANtxBuff->bufferFull) && ((IS_CANrxNew(SDO->CANrxNew)) || (SDO->state == CO_SDO_ST_UPLOAD_BL_SUBBLOCK))){
828  uint8_t CCS = SDO->CANrxData[0] >> 5; /* Client command specifier */
829  //printf("\nYOure on the right track\n");
830 
831  /* reset timeout */
832  if(SDO->state != CO_SDO_ST_UPLOAD_BL_SUBBLOCK)
833  SDO->timeoutTimer = 0;
834 
835  /* clear response buffer */
836  SDO->CANtxBuff->data[0] = SDO->CANtxBuff->data[1] = SDO->CANtxBuff->data[2] = SDO->CANtxBuff->data[3] = 0;
837  SDO->CANtxBuff->data[4] = SDO->CANtxBuff->data[5] = SDO->CANtxBuff->data[6] = SDO->CANtxBuff->data[7] = 0;
838 
839  /* Is abort from client? */
840  if((IS_CANrxNew(SDO->CANrxNew)) && (SDO->CANrxData[0] == CCS_ABORT)){
841  SDO->state = CO_SDO_ST_IDLE;
842  CLEAR_CANrxNew(SDO->CANrxNew);
843  return -1;
844  }
845  //printf("\nSDOSTATE2 Print: %u ", state);
846  /* continue with previous SDO communication or start new */
847  if(SDO->state != CO_SDO_ST_IDLE){
848  state = SDO->state;
849  }
850  else{
851  uint32_t abortCode;
852  uint16_t index;
853 
854  /* Is client command specifier valid */
855  if((CCS != CCS_DOWNLOAD_INITIATE) && (CCS != CCS_UPLOAD_INITIATE) &&
856  (CCS != CCS_DOWNLOAD_BLOCK) && (CCS != CCS_UPLOAD_BLOCK)){
857  CO_SDO_abort(SDO, CO_SDO_AB_CMD);/* Client command specifier not valid or unknown. */
858  return -1;
859  }
860  //printf("\nSDOSTATE3 Print: %u ", state);
861 
862  /* init ODF_arg */
863  index = SDO->CANrxData[2];
864  index = index << 8 | SDO->CANrxData[1];
865  //printf("\nSDOSTATE4 Print: %u ", index);
866  abortCode = CO_SDO_initTransfer(SDO, index, SDO->CANrxData[3]);
867  if(abortCode != 0U){
868  CO_SDO_abort(SDO, abortCode);
869  return -1;
870  }
871  /* download */
872  if((CCS == CCS_DOWNLOAD_INITIATE) || (CCS == CCS_DOWNLOAD_BLOCK)){
873  if((SDO->ODF_arg.attribute & CO_ODA_WRITEABLE) == 0U){
874  CO_SDO_abort(SDO, CO_SDO_AB_READONLY); /* attempt to write a read-only object */
875  return -1;
876  }
877  //printf("\nDOWNLOading Print: %u ", state);
878 
879  /* set state machine to normal or block download */
880  if(CCS == CCS_DOWNLOAD_INITIATE){
881  state = CO_SDO_ST_DOWNLOAD_INITIATE;
882  }
883  else{
884  state = CO_SDO_ST_DOWNLOAD_BL_INITIATE;
885  }
886  }
887 
888  /* upload */
889  else{
890  //printf("\nUPLOading Print: %u ", state);
891 
892  abortCode = CO_SDO_readOD(SDO, CO_SDO_BUFFER_SIZE);
893  if(abortCode != 0U){
894  CO_SDO_abort(SDO, abortCode);
895  return -1;
896  }
897  //printf("\nUPLOading2 Print: %u ", state);
898 
899  /* if data size is large enough set state machine to block upload, otherwise set to normal transfer */
900  if((CCS == CCS_UPLOAD_BLOCK) && (SDO->ODF_arg.dataLength > SDO->CANrxData[5])){
901  state = CO_SDO_ST_UPLOAD_BL_INITIATE;
902  }
903  else{
904  state = CO_SDO_ST_UPLOAD_INITIATE;
905  }
906  }
907  }
908  }
909  //printf("\nSDOSTATE5 Print: %u ", state);
910  /* verify SDO timeout */
911  if(SDO->timeoutTimer < SDOtimeoutTime){
912  SDO->timeoutTimer += timeDifference_ms;
913  }
914  if(SDO->timeoutTimer >= SDOtimeoutTime){
915  if((SDO->state == CO_SDO_ST_DOWNLOAD_BL_SUBBLOCK) && (SDO->sequence != 0) && (!SDO->CANtxBuff->bufferFull)){
916  timeoutSubblockDownolad = true;
917  state = CO_SDO_ST_DOWNLOAD_BL_SUB_RESP;
918  }
919  else{
920  CO_SDO_abort(SDO, CO_SDO_AB_TIMEOUT); /* SDO protocol timed out */
921  return -1;
922  }
923  }
924  //printf("\nSDOSTATE6 Print: %u ", state);
925 
926  /* return immediately if still idle */
927  if(state == CO_SDO_ST_IDLE){
928  return 0;
929  }
930  //printf("\nSDOSTATE7 Print: %u ", state);
931 
932  /* state machine (buffer is freed (CLEAR_CANrxNew()) at the end) */
933  switch(state){
934  uint32_t abortCode;
935  uint16_t len, i;
936  bool_t lastSegmentInSubblock;
937 
938  case CO_SDO_ST_DOWNLOAD_INITIATE:{
939  /* default response */
940  SDO->CANtxBuff->data[0] = 0x60;
941  SDO->CANtxBuff->data[1] = SDO->CANrxData[1];
942  SDO->CANtxBuff->data[2] = SDO->CANrxData[2];
943  SDO->CANtxBuff->data[3] = SDO->CANrxData[3];
944 
945  /* Expedited transfer */
946  if((SDO->CANrxData[0] & 0x02U) != 0U){
947  //printf("\nDOWNLOAD_INITIATE Print1: %u ", state);
948  /* is size indicated? Get message length */
949  if((SDO->CANrxData[0] & 0x01U) != 0U){
950  len = 4U - ((SDO->CANrxData[0] >> 2U) & 0x03U);
951  }
952  else{
953  len = SDO->ODF_arg.dataLength;
954  }
955 
956  /* copy data to SDO buffer */
957  SDO->ODF_arg.data[0] = SDO->CANrxData[4];
958  SDO->ODF_arg.data[1] = SDO->CANrxData[5];
959  SDO->ODF_arg.data[2] = SDO->CANrxData[6];
960  SDO->ODF_arg.data[3] = SDO->CANrxData[7];
961 
962  /* write data to the Object dictionary */
963  abortCode = CO_SDO_writeOD(SDO, len);
964  if(abortCode != 0U){
965  CO_SDO_abort(SDO, abortCode);
966  return -1;
967  }
968  //printf("\nDOWNLOAD_INITIATE Print2: %u ", state);
969 
970  /* finish the communication */
971  SDO->state = CO_SDO_ST_IDLE;
972  sendResponse = true;
973  }
974 
975  /* Segmented transfer */
976  else{
977  /* verify length if size is indicated */
978  if((SDO->CANrxData[0]&0x01) != 0){
979  uint32_t lenRx;
980  //printf("\nDOWNLOAD_INITIATE Print3: %u ", &SDO->CANrxData[4]);
981 
982  CO_memcpySwap4(&lenRx, &SDO->CANrxData[4]);
983 
984  SDO->ODF_arg.dataLengthTotal = lenRx;
985  //printf("\nDOWNLOAD_INITIATE Print3: %u ", lenRx);
986  //printf("\nDOWNLOAD_INITIATE Print4: %u ", SDO->ODF_arg.dataLength);
987 
988  /* verify length except for domain data type */
989  if((lenRx != SDO->ODF_arg.dataLength) && (SDO->ODF_arg.ODdataStorage != 0)){
990  CO_SDO_abort(SDO, CO_SDO_AB_TYPE_MISMATCH); /* Length of service parameter does not match */
991  return -1;
992  }
993  }
994  SDO->bufferOffset = 0;
995  SDO->sequence = 0;
996  SDO->state = CO_SDO_ST_DOWNLOAD_SEGMENTED;
997  sendResponse = true;
998  }
999  break;
1000  }
1001 
1002  case CO_SDO_ST_DOWNLOAD_SEGMENTED:{
1003  /* verify client command specifier */
1004  if((SDO->CANrxData[0]&0xE0) != 0x00U){
1005  CO_SDO_abort(SDO, CO_SDO_AB_CMD);/* Client command specifier not valid or unknown. */
1006  return -1;
1007  }
1008 
1009  /* verify toggle bit */
1010  i = (SDO->CANrxData[0]&0x10U) ? 1U : 0U;
1011  if(i != SDO->sequence){
1012  CO_SDO_abort(SDO, CO_SDO_AB_TOGGLE_BIT);/* toggle bit not alternated */
1013  return -1;
1014  }
1015 
1016  /* get size of data in message */
1017  len = 7U - ((SDO->CANrxData[0] >> 1U) & 0x07U);
1018 
1019  /* verify length. Domain data type enables length larger than SDO buffer size */
1020  if((SDO->bufferOffset + len) > SDO->ODF_arg.dataLength){
1021  if(SDO->ODF_arg.ODdataStorage != 0){
1022  CO_SDO_abort(SDO, CO_SDO_AB_DATA_LONG); /* Length of service parameter too high */
1023  return -1;
1024  }
1025  else{
1026  /* empty buffer in domain data type */
1027  SDO->ODF_arg.lastSegment = false;
1028  abortCode = CO_SDO_writeOD(SDO, SDO->bufferOffset);
1029  if(abortCode != 0U){
1030  CO_SDO_abort(SDO, abortCode);
1031  return -1;
1032  }
1033 
1034  SDO->ODF_arg.dataLength = CO_SDO_BUFFER_SIZE;
1035  SDO->bufferOffset = 0;
1036  }
1037  }
1038 
1039  /* copy data to buffer */
1040  for(i=0U; i<len; i++)
1041  SDO->ODF_arg.data[SDO->bufferOffset++] = SDO->CANrxData[i+1];
1042 
1043  /* If no more segments to be downloaded, write data to the Object dictionary */
1044  if((SDO->CANrxData[0] & 0x01U) != 0U){
1045  SDO->ODF_arg.lastSegment = true;
1046  abortCode = CO_SDO_writeOD(SDO, SDO->bufferOffset);
1047  if(abortCode != 0U){
1048  CO_SDO_abort(SDO, abortCode);
1049  return -1;
1050  }
1051 
1052  /* finish */
1053  SDO->state = CO_SDO_ST_IDLE;
1054  }
1055 
1056  /* download segment response and alternate toggle bit */
1057  SDO->CANtxBuff->data[0] = 0x20 | (SDO->sequence ? 0x10 : 0x00);
1058  SDO->sequence = (SDO->sequence) ? 0 : 1;
1059  sendResponse = true;
1060  break;
1061  }
1062 
1063  case CO_SDO_ST_DOWNLOAD_BL_INITIATE:{
1064  /* verify client command specifier and subcommand */
1065  if((SDO->CANrxData[0]&0xE1U) != 0xC0U){
1066  CO_SDO_abort(SDO, CO_SDO_AB_CMD);/* Client command specifier not valid or unknown. */
1067  return -1;
1068  }
1069 
1070  /* prepare response */
1071  SDO->CANtxBuff->data[0] = 0xA4;
1072  SDO->CANtxBuff->data[1] = SDO->CANrxData[1];
1073  SDO->CANtxBuff->data[2] = SDO->CANrxData[2];
1074  SDO->CANtxBuff->data[3] = SDO->CANrxData[3];
1075 
1076  /* blksize */
1077  SDO->blksize = (CO_SDO_BUFFER_SIZE > (7*127)) ? 127 : (CO_SDO_BUFFER_SIZE / 7);
1078  SDO->CANtxBuff->data[4] = SDO->blksize;
1079 
1080  /* is CRC enabled */
1081  SDO->crcEnabled = (SDO->CANrxData[0] & 0x04) ? true : false;
1082  SDO->crc = 0;
1083 
1084  /* verify length if size is indicated */
1085  if((SDO->CANrxData[0]&0x02) != 0U){
1086  uint32_t lenRx;
1087  CO_memcpySwap4(&lenRx, &SDO->CANrxData[4]);
1088  SDO->ODF_arg.dataLengthTotal = lenRx;
1089 
1090  /* verify length except for domain data type */
1091  if((lenRx != SDO->ODF_arg.dataLength) && (SDO->ODF_arg.ODdataStorage != 0)){
1092  CO_SDO_abort(SDO, CO_SDO_AB_TYPE_MISMATCH); /* Length of service parameter does not match */
1093  return -1;
1094  }
1095  }
1096 
1097  SDO->bufferOffset = 0;
1098  SDO->sequence = 0;
1099  SDO->state = CO_SDO_ST_DOWNLOAD_BL_SUBBLOCK;
1100 
1101  /* send response */
1102  sendResponse = true;
1103  break;
1104  }
1105 
1106  case CO_SDO_ST_DOWNLOAD_BL_SUBBLOCK:{
1107  /* data are copied directly in receive function */
1108  break;
1109  }
1110 
1111  case CO_SDO_ST_DOWNLOAD_BL_SUB_RESP:{
1112  /* no new message received, SDO timeout occured, try to response */
1113  lastSegmentInSubblock = (!timeoutSubblockDownolad &&
1114  ((SDO->CANrxData[0] & 0x80U) == 0x80U)) ? true : false;
1115 
1116  /* prepare response */
1117  SDO->CANtxBuff->data[0] = 0xA2;
1118  SDO->CANtxBuff->data[1] = SDO->sequence;
1119  SDO->sequence = 0;
1120 
1121  /* empty buffer in domain data type if not last segment */
1122  if((SDO->ODF_arg.ODdataStorage == 0) && (SDO->bufferOffset != 0) && !lastSegmentInSubblock){
1123  /* calculate CRC on next bytes, if enabled */
1124  if(SDO->crcEnabled){
1125  SDO->crc = crc16_ccitt(SDO->ODF_arg.data, SDO->bufferOffset, SDO->crc);
1126  }
1127 
1128  /* write data to the Object dictionary */
1129  SDO->ODF_arg.lastSegment = false;
1130  abortCode = CO_SDO_writeOD(SDO, SDO->bufferOffset);
1131  if(abortCode != 0U){
1132  CO_SDO_abort(SDO, abortCode);
1133  return -1;
1134  }
1135 
1136  SDO->ODF_arg.dataLength = CO_SDO_BUFFER_SIZE;
1137  SDO->bufferOffset = 0;
1138  }
1139 
1140  /* blksize */
1141  len = CO_SDO_BUFFER_SIZE - SDO->bufferOffset;
1142  SDO->blksize = (len > (7*127)) ? 127 : (len / 7);
1143  SDO->CANtxBuff->data[2] = SDO->blksize;
1144 
1145  /* set next state */
1146  if(lastSegmentInSubblock) {
1147  SDO->state = CO_SDO_ST_DOWNLOAD_BL_END;
1148  }
1149  else if(SDO->bufferOffset >= CO_SDO_BUFFER_SIZE) {
1150  CO_SDO_abort(SDO, CO_SDO_AB_DEVICE_INCOMPAT);
1151  return -1;
1152  }
1153  else {
1154  SDO->state = CO_SDO_ST_DOWNLOAD_BL_SUBBLOCK;
1155  }
1156 
1157  /* send response */
1158  sendResponse = true;
1159 
1160  break;
1161  }
1162 
1163  case CO_SDO_ST_DOWNLOAD_BL_END:{
1164  /* verify client command specifier and subcommand */
1165  if((SDO->CANrxData[0]&0xE1U) != 0xC1U){
1166  CO_SDO_abort(SDO, CO_SDO_AB_CMD);/* Client command specifier not valid or unknown. */
1167  return -1;
1168  }
1169 
1170  /* number of bytes in the last segment of the last block that do not contain data. */
1171  len = (SDO->CANrxData[0]>>2U) & 0x07U;
1172  SDO->bufferOffset -= len;
1173 
1174  /* calculate and verify CRC, if enabled */
1175  if(SDO->crcEnabled){
1176  uint16_t crc;
1177  SDO->crc = crc16_ccitt(SDO->ODF_arg.data, SDO->bufferOffset, SDO->crc);
1178 
1179  CO_memcpySwap2(&crc, &SDO->CANrxData[1]);
1180 
1181  if(SDO->crc != crc){
1182  CO_SDO_abort(SDO, CO_SDO_AB_CRC); /* CRC error (block mode only). */
1183  return -1;
1184  }
1185  }
1186 
1187  /* write data to the Object dictionary */
1188  SDO->ODF_arg.lastSegment = true;
1189  abortCode = CO_SDO_writeOD(SDO, SDO->bufferOffset);
1190  if(abortCode != 0U){
1191  CO_SDO_abort(SDO, abortCode);
1192  return -1;
1193  }
1194 
1195  /* send response */
1196  SDO->CANtxBuff->data[0] = 0xA1;
1197  SDO->state = CO_SDO_ST_IDLE;
1198  sendResponse = true;
1199  break;
1200  }
1201 
1202  case CO_SDO_ST_UPLOAD_INITIATE:{
1203  /* default response */
1204  //printf("\n CO_SDO_ST_UPLOAD_INITIATE1");
1205 
1206  SDO->CANtxBuff->data[1] = SDO->CANrxData[1];
1207  SDO->CANtxBuff->data[2] = SDO->CANrxData[2];
1208  SDO->CANtxBuff->data[3] = SDO->CANrxData[3];
1209 
1210  /* Expedited transfer */
1211  if(SDO->ODF_arg.dataLength <= 4U){
1212  //printf("\n CO_SDO_ST_UPLOAD_INITIATE2 \n");
