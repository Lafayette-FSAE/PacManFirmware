33 /*
34  * Read received message from CAN module.
35  *
36  * Function will be called (by CAN receive interrupt) every time, when CAN
37  * message with correct identifier will be received. For more information and
38  * description of parameters see file CO_driver.h.

40 static void CO_EM_receive(void *object, const CO_CANrxMsg_t *msg){
41  CO_EM_t *em;
42  uint16_t errorCode;
43  uint32_t infoCode;
44 
45  em = (CO_EM_t*)object;
46 
47  if(em!=NULL && em->pFunctSignalRx!=NULL){
48  CO_memcpySwap2(&errorCode, &msg->data[0]);
49  CO_memcpySwap4(&infoCode, &msg->data[4]);
50  em->pFunctSignalRx(CO_CANrxMsg_readIdent(msg),
51  errorCode,
52  msg->data[2],
53  msg->data[3],
54  infoCode);
55  }
56 }

60  *
61  * For more information see file CO_SDO.h.
62  */
63 static CO_SDO_abortCode_t CO_ODF_1003(CO_ODF_arg_t *ODF_arg);
64 static CO_SDO_abortCode_t CO_ODF_1003(CO_ODF_arg_t *ODF_arg){
65  CO_EMpr_t *emPr;
66  uint8_t value;
67  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
68 
69  emPr = (CO_EMpr_t*) ODF_arg->object;
70  value = ODF_arg->data[0];
71 
72  if(ODF_arg->reading){
73  uint8_t noOfErrors;
74  noOfErrors = emPr->preDefErrNoOfErrors;
75 
76  if(ODF_arg->subIndex == 0U){
77  ODF_arg->data[0] = noOfErrors;
78  }
79  else if(ODF_arg->subIndex > noOfErrors){
80  ret = CO_SDO_AB_NO_DATA;
81  }
82  else{
83  ret = CO_SDO_AB_NONE;
84  }
85  }
86  else{
87  /* only '0' may be written to subIndex 0 */
88  if(ODF_arg->subIndex == 0U){
89  if(value == 0U){
90  emPr->preDefErrNoOfErrors = 0U;
91  }
92  else{
93  ret = CO_SDO_AB_INVALID_VALUE;
94  }
95  }
96  else{
97  ret = CO_SDO_AB_READONLY;
98  }
99  }
100 
101  return ret;
102 }

128 /******************************************************************************/
129 CO_ReturnError_t CO_EM_init(
130  CO_EM_t *em,
131  CO_EMpr_t *emPr,
132  CO_SDO_t *SDO,
133  uint8_t *errorStatusBits,
134  uint8_t errorStatusBitsSize,
135  uint8_t *errorRegister,
136  uint32_t *preDefErr,
137  uint8_t preDefErrSize,
138  CO_CANmodule_t *CANdevRx,
139  uint16_t CANdevRxIdx,
140  CO_CANmodule_t *CANdevTx,
141  uint16_t CANdevTxIdx,
142  uint16_t CANidTxEM)
143 {
144  uint8_t i;
145 
146  /* verify arguments */
147  if(em==NULL || emPr==NULL || SDO==NULL || errorStatusBits==NULL || errorStatusBitsSize<6U ||
148  errorRegister==NULL || preDefErr==NULL || CANdevTx==NULL || CANdevRx==NULL){
149  return CO_ERROR_ILLEGAL_ARGUMENT;
150  }
151 
152  /* Configure object variables */
153  em->errorStatusBits = errorStatusBits;
154  em->errorStatusBitsSize = errorStatusBitsSize;
155  em->bufEnd = em->buf + (CO_EM_INTERNAL_BUFFER_SIZE * 8);
156  em->bufWritePtr = em->buf;
157  em->bufReadPtr = em->buf;
158  em->bufFull = 0U;
159  em->wrongErrorReport = 0U;
160  em->pFunctSignal = NULL;
161  em->pFunctSignalRx = NULL;
162  emPr->em = em;
163  emPr->errorRegister = errorRegister;
164  emPr->preDefErr = preDefErr;
165  emPr->preDefErrSize = preDefErrSize;
166  emPr->preDefErrNoOfErrors = 0U;
167  emPr->inhibitEmTimer = 0U;
168 
169  /* clear error status bits */
170  for(i=0U; i<errorStatusBitsSize; i++){
171  em->errorStatusBits[i] = 0U;
172  }
173 
174  /* Configure Object dictionary entry at index 0x1003 and 0x1014 */
175  CO_OD_configure(SDO, OD_H1003_PREDEF_ERR_FIELD, CO_ODF_1003, (void*)emPr, 0, 0U);
176  CO_OD_configure(SDO, OD_H1014_COBID_EMERGENCY, CO_ODF_1014, (void*)&SDO->nodeId, 0, 0U);
177 
178  /* configure SDO server CAN reception */
179  printf("\nCO_CAN_ID_EMERGENCY Print: %u ", CANdevRxIdx);
180  CO_CANrxBufferInit(
181  CANdevRx, /* CAN device */
182  CANdevRxIdx, /* rx buffer index */
183  CO_CAN_ID_EMERGENCY, /* CAN identifier */
184  0x780, /* mask */
185  0, /* rtr */
186  (void*)em, /* object passed to receive function */
187  CO_EM_receive); /* this function will process received message */
188 
189  /* configure emergency message CAN transmission */
190  emPr->CANdev = CANdevTx;
191  emPr->CANdev->em = (void*)em; /* update pointer inside CAN device. */
192  emPr->CANtxBuff = CO_CANtxBufferInit(
193  CANdevTx, /* CAN device */
194  CANdevTxIdx, /* index of specific buffer inside CAN module */
195  CANidTxEM, /* CAN identifier */
196  0, /* rtr */
197  8U, /* number of data bytes */
198  0); /* synchronous message flag bit */
199 
200  return CO_ERROR_NO;
201 }

230 /******************************************************************************/
231 void CO_EM_process(
232  CO_EMpr_t *emPr,
233  bool_t NMTisPreOrOperational,
234  uint16_t timeDifference_100us,
235  uint16_t emInhTime,
236  uint16_t *timerNext_ms)
237 {
238 
239  CO_EM_t *em = emPr->em;
240  uint8_t errorRegister;
241  uint8_t errorMask;
242  uint8_t i;
243 
244  /* verify errors from driver and other */
245  CO_CANverifyErrors(emPr->CANdev);
246  if(em->wrongErrorReport != 0U){
247  CO_errorReport(em, CO_EM_WRONG_ERROR_REPORT, CO_EMC_SOFTWARE_INTERNAL, (uint32_t)em->wrongErrorReport);
248  em->wrongErrorReport = 0U;
249  }
250 
251 
252  /* calculate Error register */
253  errorRegister = 0U;
254  errorMask = (uint8_t)~(CO_ERR_REG_GENERIC_ERR | CO_ERR_REG_COMM_ERR | CO_ERR_REG_MANUFACTURER);
255  /* generic error */
256  if(em->errorStatusBits[5]){
257  errorRegister |= CO_ERR_REG_GENERIC_ERR;
258  }
259  /* communication error (overrun, error state) */
260  if(em->errorStatusBits[2] || em->errorStatusBits[3]){
261  errorRegister |= CO_ERR_REG_COMM_ERR;
262  }
263  /* Manufacturer */
264  for(i=6; i<em->errorStatusBitsSize; i++) {
265  if (em->errorStatusBits[i]) {
266  errorRegister |= CO_ERR_REG_MANUFACTURER;
267  }
268  }
269  *emPr->errorRegister = (*emPr->errorRegister & errorMask) | errorRegister;
270 
271  /* inhibit time */
272  if(emPr->inhibitEmTimer < emInhTime){
273  emPr->inhibitEmTimer += timeDifference_100us;
274  }
275 
276  /* send Emergency message. */
277  if( NMTisPreOrOperational &&
278  !emPr->CANtxBuff->bufferFull &&
279  (em->bufReadPtr != em->bufWritePtr || em->bufFull))
280  {
281  uint32_t preDEF; /* preDefinedErrorField */
282  uint16_t diff;
283 
284  if (emPr->inhibitEmTimer >= emInhTime) {
285  /* inhibit time elapsed, send message */
286 
287  /* add error register */
288  em->bufReadPtr[2] = *emPr->errorRegister;
289 
290  /* copy data to CAN emergency message */
291  CO_memcpy(emPr->CANtxBuff->data, em->bufReadPtr, 8U);
292  CO_memcpy((uint8_t*)&preDEF, em->bufReadPtr, 4U);
293  em->bufReadPtr += 8;
294 
295  /* Update read buffer pointer and reset inhibit timer */
296  if(em->bufReadPtr == em->bufEnd){
297  em->bufReadPtr = em->buf;
298  }
299  emPr->inhibitEmTimer = 0U;
300 
301  /* verify message buffer overflow, then clear full flag */
302  if(em->bufFull == 2U){
303  em->bufFull = 0U; /* will be updated below */
304  CO_errorReport(em, CO_EM_EMERGENCY_BUFFER_FULL, CO_EMC_GENERIC, 0U);
305  }
306  else{
307  em->bufFull = 0;
308  CO_errorReset(em, CO_EM_EMERGENCY_BUFFER_FULL, 0);
309  }
310 
311  /* write to 'pre-defined error field' (object dictionary, index 0x1003) */
312  if(emPr->preDefErr){
313  uint8_t j;
314 
315  if(emPr->preDefErrNoOfErrors < emPr->preDefErrSize)
316  emPr->preDefErrNoOfErrors++;
317  for(j=emPr->preDefErrNoOfErrors-1; j>0; j--)
318  emPr->preDefErr[j] = emPr->preDefErr[j-1];
319  emPr->preDefErr[0] = preDEF;
320  }
321 
322  /* send CAN message */
323  CO_CANsend(emPr->CANdev, emPr->CANtxBuff);
324  }
325 
326  /* check again after inhibit time elapsed */
327  diff = (emInhTime + 9) / 10; /* time difference in ms, always round up */
328  if (timerNext_ms != NULL && *timerNext_ms > diff) {
329  *timerNext_ms = diff;
330  }
331  }
332 
333  return;
334 }

337 /******************************************************************************/
338 void CO_errorReport(CO_EM_t *em, const uint8_t errorBit, const uint16_t errorCode, const uint32_t infoCode){
339  uint8_t index = errorBit >> 3;
340  uint8_t bitmask = 1 << (errorBit & 0x7);
341  uint8_t *errorStatusBits = 0;
342  bool_t sendEmergency = true;
343 
344  if(em == NULL){
345  sendEmergency = false;
346  }
347  else if(index >= em->errorStatusBitsSize){
348  /* if errorBit value not supported, send emergency 'CO_EM_WRONG_ERROR_REPORT' */
349  em->wrongErrorReport = errorBit;
350  sendEmergency = false;
351  }
352  else{
353  errorStatusBits = &em->errorStatusBits[index];
354  /* if error was already reported, do nothing */
355  if((*errorStatusBits & bitmask) != 0){
356  sendEmergency = false;
357  }
358  }
359 
360  if(sendEmergency){
361  /* set error bit */
362  if(errorBit){
363  /* any error except NO_ERROR */
364  *errorStatusBits |= bitmask;
365  }
366 
367  /* verify buffer full, set overflow */
368  if(em->bufFull){
369  em->bufFull = 2;
370  }
371  else{
372  uint8_t bufCopy[8];
373 
374  /* prepare data for emergency message */
375  CO_memcpySwap2(&bufCopy[0], &errorCode);
376  bufCopy[2] = 0; /* error register will be set later */
377  bufCopy[3] = errorBit;
378  CO_memcpySwap4(&bufCopy[4], &infoCode);
379 
380  /* copy data to the buffer, increment writePtr and verify buffer full */
381  CO_LOCK_EMCY();
382  CO_memcpy(em->bufWritePtr, &bufCopy[0], 8);
383  em->bufWritePtr += 8;
384 
385  if(em->bufWritePtr == em->bufEnd) em->bufWritePtr = em->buf;
386  if(em->bufWritePtr == em->bufReadPtr) em->bufFull = 1;
387  CO_UNLOCK_EMCY();
388 
389  /* Optional signal to RTOS, which can resume task, which handles CO_EM_process */
390  if(em->pFunctSignal != NULL) {
391  em->pFunctSignal();
392  }
393  }
394  }
395 }
